\title{Gravity from Quantum Measurement}
\author[ ]{Kevin Player\footnote{\textit{kjplaye@gmail.com}} \,\orcidlink{0009-0000-7180-7985}}

\abstract{We propose an operational framework in which gravity emerges from the physical backreaction associated with quantum measurement. Starting from a commutative structure linking state reduction, observation, and acceleration, we combine Landauer’s principle with relativistic energy–momentum transfer to argue that each act of measurement necessarily imparts a discrete impulse to the observer. By the equivalence principle, this measurement-induced acceleration is locally indistinguishable from a gravitational field. At the level of a single bit, this correspondence naturally identifies the Schwarzschild radius as the scale at which information transfer produces gravitational-strength effects. Extending to many measurements within a finite region, the total energy associated with information acquisition is bounded by gravitational collapse, leading directly to the Bekenstein entropy bound and its area scaling. In this way, Newtonian gravity and black hole thermodynamics arise from a common underlying principle: that information is physically mediated and that its transfer carries unavoidable energy–momentum consequences. This perspective contrasts with entropic gravity approaches by grounding the origin of gravitational effects in local, observer-dependent measurement processes rather than global entropy gradients.}


\section{Measurement-Induced Gravity}

We propose that gravitational effects arise from the physical backreaction associated with quantum measurement. The argument is organized around the operational chain
\begin{equation}
R \;\longrightarrow\; O \;\longrightarrow\; A \;\longleftrightarrow\; G,
\end{equation}
where $R$ denotes quantum state reduction, $O$ observation (information extraction), $A$ acceleration of the observer, and $G$ gravity via the equivalence principle.

\subsection{Information and Energy}

A measurement extracts at least one bit of information,
\begin{equation}
\Delta I = 1,
\end{equation}
which by Landauer's principle requires a minimal energy cost
\begin{equation}
\Delta E \;\ge\; k_B T \ln 2.
\end{equation}
We interpret this energy as carried by a physical quantum (a ``thrust particle'') with effective mass
\begin{equation}
\Delta E = m_1 c^2.
\end{equation}

\subsection{Impulse and Acceleration}

The transfer of energy $\Delta E$ implies a momentum transfer
\begin{equation}
\Delta p = \frac{\Delta E}{c}.
\end{equation}
For a detector of mass $m_2$, this produces an acceleration over a time interval $\Delta t$:
\begin{equation}
a = \frac{\Delta p}{m_2 \Delta t} = \frac{\Delta E}{c\, m_2 \Delta t}.
\end{equation}

Assuming the interaction occurs over a spatial scale $r$, we take
\begin{equation}
\Delta t \sim \frac{r}{c},
\end{equation}
so that
\begin{equation}
a = \frac{\Delta E}{m_2 r}.
\end{equation}
Substituting $\Delta E = m_1 c^2$ gives
\begin{equation}
a = \frac{m_1 c^2}{m_2 r}.
\end{equation}

\subsection{Emergence of Gravitational Scaling}

Comparing with Newtonian gravity,
\begin{equation}
a = \frac{G m_1}{r^2},
\end{equation}
we obtain the relation
\begin{equation}
G \sim \frac{c^2 r}{m_2}.
\end{equation}
Rewriting,
\begin{equation}
r \sim \frac{G m_2}{c^2},
\end{equation}
which identifies the interaction scale with the Schwarzschild radius of the detector mass $m_2$.

\subsection{Many Bits and Energy Bounds}

For $N$ independent measurements,
\begin{equation}
E_{\text{tot}} = N \Delta E.
\end{equation}
To avoid gravitational collapse within a region of size $R$, the total energy must satisfy
\begin{equation}
E_{\text{tot}} \;\lesssim\; \frac{c^4 R}{2G}.
\end{equation}
This gives an upper bound on the number of bits:
\begin{equation}
N \;\lesssim\; \frac{c^4 R}{2G \Delta E}.
\end{equation}

\subsection{Entropy Bound}

The corresponding entropy is
\begin{equation}
S = N k_B \ln 2 \;\lesssim\; \frac{c^4 R}{2G T}.
\end{equation}
Using the Unruh relation between acceleration and temperature,
\begin{equation}
T \sim \frac{\hbar a}{2\pi k_B c},
\end{equation}
and evaluating at the horizon scale $a \sim c^2 / R$, we obtain
\begin{equation}
T \sim \frac{\hbar c}{2\pi k_B R}.
\end{equation}
Substituting gives
\begin{equation}
S \sim \frac{c^3 R^2}{\hbar G},
\end{equation}
which reproduces the Bekenstein--Hawking scaling
\begin{equation}
S \sim \frac{A}{\ell_P^2}, \qquad \ell_P^2 = \frac{\hbar G}{c^3}.
\end{equation}

\subsection{Summary}

We thus obtain the chain
\begin{equation}
\text{measurement} \;\to\; \Delta E \;\to\; \Delta p \;\to\; a \;\leftrightarrow\; g \;\to\; r_s \;\to\; S \sim A,
\end{equation}
linking quantum state reduction to gravitational dynamics and horizon entropy.
